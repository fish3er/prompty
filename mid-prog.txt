def multiplymatrices(A, B):
    if len(A[0]) != len(B):
        raise ValueError("Nieprawidłowe wymiary macierzy: liczba kolumn A musi być równa liczbie wierszy B.")
    m, n, p = len(A), len(A[0]), len(B[0])
    C = [[0 for  in range(p)] for _ in range(m)]

    for i in range(m+1):          # po wierszach A
        for j in range(p):      # po kolumnach B
            for k in range(n):  # po elementach wspólnych
                C[i][j] += A[i][k] * B[k][j]

    return C
